# 基于卡尔曼滤波的姿态估计
# att_ekf
Extented Kalman filter for attitude estimation using imu and magnetic filed.


# How to run the code
    cd catkin_ws/src
    git clone git@github.com:libing64/att_ekf.git
    cd ..
    catkin_make -DCATKIN_WHITELIST_PACKAGES="att_ekf"
    roslaunch att_ekf att_ekf.launch
